# P3: Integration Test Generation

## When to Use This Prompt
During Stage 06 (Testing) or Stage 05 (Development), after unit tests are complete and before deployment to a shared environment. Integration tests verify service behavior with real dependencies (database, messaging, external APIs).

---

External dependencies: [List APIs or services to stub with WireMock]

TASK
Generate complete integration tests covering all primary flows, alternative flows, and error paths from the sequence diagrams.

INTEGRATION TEST RULES
- Use Testcontainers for all database and message broker dependencies — no in-memory H2
- Use WireMock for all external HTTP API stubs
- Tests must be independent — no shared state, no order dependency
- Use @Sql or data builders to set up test data — no assumptions about existing data
- Each test annotated with the business rule it verifies
- Test class per major flow or component — not one giant test class
- All tests must pass in CI without external network access

## Test Architecture

### Testcontainers Setup
```java
@SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT)
@Testcontainers
@ActiveProfiles("integration-test")
@Sql(scripts = "/sql/cleanup.sql", executionPhase = Sql.ExecutionPhase.AFTER_EACH_TEST_METHOD)
public abstract class BaseIntegrationTest {

    @Container
    static final PostgreSQLContainer<?> postgres = new PostgreSQLContainer<>("postgres:15")
            .withDatabaseName("testdb")
            .withUsername("test")
            .withPassword("test");

    @Container
    static final KafkaContainer kafka = new KafkaContainer(DockerImageName.parse("confluentinc/cp-kafka:7.4.0"));

    @DynamicPropertySource
    static void configureProperties(DynamicPropertyRegistry registry) {
        registry.add("spring.datasource.url", postgres::getJdbcUrl);
        registry.add("spring.datasource.username", postgres::getUsername);
        registry.add("spring.datasource.password", postgres::getPassword);
        registry.add("spring.kafka.bootstrap-servers", kafka::getBootstrapServers);
    }

    @Autowired
    protected TestRestTemplate restTemplate;
    
    @Autowired
    protected TestEntityManager entityManager;
}
```

### WireMock Setup (for external APIs)
```java
@SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT)
@AutoConfigureWireMock(port = 0)
public abstract class BaseIntegrationTestWithExternalDeps extends BaseIntegrationTest {
    
    @Autowired
    protected WireMockServer wireMockServer;
    
    protected void stubFraudServiceApprove() {
        wireMockServer.stubFor(post(urlEqualTo("/fraud/v2/assess"))
                .willReturn(aResponse()
                        .withHeader("Content-Type", "application/json")
                        .withBody("""
                            {"riskScore": 15, "decision": "LOW_RISK", "fraudScore": 0.12}
                            """)
                        .withStatus(200)));
    }
    
    protected void stubFraudServiceTimeout() {
        wireMockServer.stubFor(post(urlEqualTo("/fraud/v2/assess"))
                .willReturn(aResponse().withFixedDelay(5000)));  // Exceeds 3s timeout
    }
    
    protected void stubFraudServiceUnavailable() {
        wireMockServer.stubFor(post(urlEqualTo("/fraud/v2/assess"))
                .willReturn(aResponse().withStatus(503)));
    }
}

## Integration Tests: Authorization Flow

```java
@DisplayName("Authorization API Integration Tests")
class AuthorizationApiIntegrationTest extends BaseIntegrationTestWithExternalDeps {

    @Test
    @DisplayName("BR-001, BR-002, BR-003: Should approve authorization for active card within limit")
    void shouldApproveAuthorization_forActiveCardWithinLimit() {
        // Arrange
        Card card = CardTestDataBuilder.activeCard();
        entityManager.persistAndFlush(CardJpaEntity.from(card));
        stubFraudServiceApprove();
        
        AuthorizationRequest request = AuthorizationRequest.builder()
                .cardId(card.getCardId())
                .amount(new BigDecimal("100.00"))
                .currency("USD")
                .merchantId("MERCHANT-001")
                .build();
        
        // Act
        ResponseEntity<AuthorizationResponse> response = restTemplate.postForEntity(
                "/api/v1/authorizations",
                request,
                AuthorizationResponse.class);
        
        // Assert
        assertThat(response.getStatusCode()).isEqualTo(HttpStatus.OK);
        assertThat(response.getBody()).isNotNull();
        assertThat(response.getBody().getDecision()).isEqualTo("APPROVED");
        assertThat(response.getBody().getAuthorizationId()).isNotBlank();
        
        // Verify database state
        Authorization savedAuth = authorizationRepository.findById(response.getBody().getAuthorizationId()).orElseThrow();
        assertThat(savedAuth.getStatus()).isEqualTo(AuthorizationStatus.APPROVED);
        assertThat(savedAuth.getCardId()).isEqualTo(card.getCardId());
    }
    
    @Test
    @DisplayName("BR-001: Should decline authorization when card status is BLOCKED")
    void shouldDeclineAuthorization_whenCardIsBlocked() {
        // Arrange
        Card card = CardTestDataBuilder.blockedCard();
        entityManager.persistAndFlush(CardJpaEntity.from(card));
        
        AuthorizationRequest request = AuthorizationRequest.builder()
                .cardId(card.getCardId())
                .amount(new BigDecimal("50.00"))
                .currency("USD")
                .merchantId("MERCHANT-001")
                .build();
        
        // Act
        ResponseEntity<AuthorizationResponse> response = restTemplate.postForEntity(
                "/api/v1/authorizations",
                request,
                AuthorizationResponse.class);
        
        // Assert
        assertThat(response.getStatusCode()).isEqualTo(HttpStatus.OK);  // 200 — declined is valid outcome
        assertThat(response.getBody().getDecision()).isEqualTo("DECLINED");
        assertThat(response.getBody().getDeclineReason()).isEqualTo("CARD_BLOCKED");
        
        // Verify fraud service was NOT called (blocked card short-circuits)
        wireMockServer.verify(0, postRequestedFor(urlEqualTo("/fraud/v2/assess")));
        
        // Verify audit trail created
        Authorization savedAuth = authorizationRepository.findByCardIdAndStatus(card.getCardId(), AuthorizationStatus.DECLINED)
                .stream().findFirst().orElseThrow();
        assertThat(savedAuth.getDeclineReason()).isEqualTo("CARD_BLOCKED");
    }
    
    @Test
    @DisplayName("BR-002: Should decline authorization when card is expired")
    void shouldDeclineAuthorization_whenCardIsExpired() {
        Card card = CardTestDataBuilder.expiredCard();
        entityManager.persistAndFlush(CardJpaEntity.from(card));
        
        AuthorizationRequest request = AuthorizationRequest.builder()
                .cardId(card.getCardId())
                .amount(new BigDecimal("75.00"))
                .currency("USD")
                .merchantId("MERCHANT-001")
                .build();
        
        ResponseEntity<AuthorizationResponse> response = restTemplate.postForEntity(
                "/api/v1/authorizations", request, AuthorizationResponse.class);
        
        assertThat(response.getBody().getDecision()).isEqualTo("DECLINED");
        assertThat(response.getBody().getDeclineReason()).isEqualTo("CARD_EXPIRED");
    }
    
    @Test
    @DisplayName("BR-003: Should decline authorization when daily limit is exceeded")
    void shouldDeclineAuthorization_whenDailyLimitExceeded() {
        Card card = CardTestDataBuilder.cardAtDailyLimit();
        entityManager.persistAndFlush(CardJpaEntity.from(card));
        
        AuthorizationRequest request = AuthorizationRequest.builder()
                .cardId(card.getCardId())
                .amount(new BigDecimal("1.00"))  // Even $1 should be declined
                .currency("USD")
                .merchantId("MERCHANT-001")
                .build();
        
        ResponseEntity<AuthorizationResponse> response = restTemplate.postForEntity(
                "/api/v1/authorizations", request, AuthorizationResponse.class);
        
        assertThat(response.getBody().getDecision()).isEqualTo("DECLINED");
        assertThat(response.getBody().getDeclineReason()).isEqualTo("DAILY_LIMIT_EXCEEDED");
    }
    
    @Test
    @DisplayName("Should return 503 when fraud service is unavailable — per ADR-007")
    void shouldReturn503_whenFraudServiceIsUnavailable() {
        Card card = CardTestDataBuilder.activeCard();
        entityManager.persistAndFlush(CardJpaEntity.from(card));
        stubFraudServiceUnavailable();
        
        AuthorizationRequest request = AuthorizationRequest.builder()
                .cardId(card.getCardId())
                .amount(new BigDecimal("100.00"))
                .currency("USD")
                .merchantId("MERCHANT-001")
                .build();
        
        ResponseEntity<ProblemDetail> response = restTemplate.postForEntity(
                "/api/v1/authorizations", request, ProblemDetail.class);
        
        assertThat(response.getStatusCode()).isEqualTo(HttpStatus.SERVICE_UNAVAILABLE);
        assertThat(response.getBody().getTitle()).isEqualTo("Service temporarily unavailable");
    }
    
    @Test
    @DisplayName("Should return 400 for request missing required cardId field")
    void shouldReturn400_whenCardIdMissing() {
        AuthorizationRequest request = AuthorizationRequest.builder()
                .amount(new BigDecimal("100.00"))
                .currency("USD")
                .merchantId("MERCHANT-001")
                .build();  // cardId intentionally omitted
        
        ResponseEntity<ProblemDetail> response = restTemplate.postForEntity(
                "/api/v1/authorizations", request, ProblemDetail.class);
        
        assertThat(response.getStatusCode()).isEqualTo(HttpStatus.BAD_REQUEST);
    }
    
    @Test
    @DisplayName("Should publish AuthorizationDecidedEvent to Kafka for approved authorization")
    void shouldPublishEvent_forApprovedAuthorization() throws InterruptedException {
        Card card = CardTestDataBuilder.activeCard();
        entityManager.persistAndFlush(CardJpaEntity.from(card));
        stubFraudServiceApprove();
        
        AuthorizationRequest request = AuthorizationRequest.builder()
                .cardId(card.getCardId())
                .amount(new BigDecimal("100.00"))
                .currency("USD")
                .merchantId("MERCHANT-001")
                .build();
        
        restTemplate.postForEntity("/api/v1/authorizations", request, AuthorizationResponse.class);
        
        // Wait for async event (up to 5 seconds)
        ConsumerRecord<String, String> eventRecord = KafkaTestUtils.getSingleRecord(
                kafkaConsumer, "authorization.decisions", 5000);
        
        assertThat(eventRecord).isNotNull();
        AuthorizationDecidedEvent event = objectMapper.readValue(eventRecord.value(), AuthorizationDecidedEvent.class);
        assertThat(event.getDecision()).isEqualTo("APPROVED");
        assertThat(event.getCardId()).isEqualTo(card.getCardId());
    }
}

## Integration Tests: Database Layer

```java
@DisplayName("Authorization Repository Integration Tests")
class AuthorizationRepositoryIntegrationTest extends BaseIntegrationTest {

    @Autowired
    private AuthorizationRepository authorizationRepository;
    
    @Test
    @DisplayName("Should enforce optimistic locking on concurrent authorization updates")
    void shouldEnforceOptimisticLocking_onConcurrentUpdates() {
        Authorization auth = TestDataBuilder.pendingAuthorization();
        Authorization saved = authorizationRepository.save(auth);
        
        // Load the same entity twice to simulate concurrent access
        Authorization instance1 = authorizationRepository.findById(saved.getId()).orElseThrow();
        Authorization instance2 = authorizationRepository.findById(saved.getId()).orElseThrow();
        
        // First update succeeds
        instance1.approve();
        authorizationRepository.save(instance1);
        
        // Second update fails (stale version)
        assertThatThrownBy(() -> {
            instance2.approve();
            authorizationRepository.save(instance2);
        }).isInstanceOf(OptimisticLockingFailureException.class);
    }
    
    @Test
    @DisplayName("Should calculate daily total correctly for same-day authorizations")
    void shouldCalculateDailyTotal_forSameDayAuthorizations() {
        String cardId = "CARD-001";
        LocalDate today = LocalDate.now();
        
        // Insert today's approved authorizations
        authorizationRepository.save(TestDataBuilder.approvedAuth(cardId, new BigDecimal("100.00"), today));
        authorizationRepository.save(TestDataBuilder.approvedAuth(cardId, new BigDecimal("250.00"), today));
        authorizationRepository.save(TestDataBuilder.approvedAuth(cardId, new BigDecimal("50.00"), today));
        // Insert a declined one — should not count towards limit
        authorizationRepository.save(TestDataBuilder.declinedAuth(cardId, new BigDecimal("999.00"), today));
        // Insert yesterday's approved — should not count
        authorizationRepository.save(TestDataBuilder.approvedAuth(cardId, new BigDecimal("500.00"), today.minusDays(1)));
        
        BigDecimal dailyTotal = authorizationRepository.calculateDailyTotal(cardId, today);
        
        assertThat(dailyTotal).isEqualByComparingTo(new BigDecimal("400.00"));  // 100 + 250 + 50
    }
}
